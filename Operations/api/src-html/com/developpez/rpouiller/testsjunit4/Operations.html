038    /**
039     * diviser successivement des long (requiert au moins 2 nombres)
040     */ 
041    public static long diviser(final long...pNombres) {
042        if(pNombres.length < 2) {
043            throw new IllegalArgumentException(
044                    "Il faut au moins deux nombres en entree");
045        }
046        long lRetour = pNombres[0];
047        for(int i=1;i<pNombres.length;i++) {
048            lRetour /= pNombres[i];
049        }
050        return lRetour;
051    }

053    /**
054     * soustraire successivement des doubles (requiert au moins 2 nombres)
055     */    
056    public static double soustraire(final double...pNombres) {
057        if(pNombres.length < 2) {
058            throw new IllegalArgumentException(
059                    "Il faut au moins deux nombres en entree");
060        }
061        double lRetour = pNombres[0];
062        for(int i=1;i<pNombres.length;i++) {
063            lRetour -= pNombres[i];
064        }
065        return lRetour;
066    }

068    /**
069     * Cette methode verifie que les longueurs passees en parametre
070     * sont celles d'un triangle rectangle
071     */
072    public static Boolean[] pythagore(final long[]...pLongueurs) {
073        final Boolean[] lRetours = new Boolean[pLongueurs.length];
074
075        for(int i=0;i<pLongueurs.length;i++) {
076            final long[] lLongeurs = pLongueurs[i];
077            if(lLongeurs.length != 3) {
078                throw new IllegalArgumentException(
079                    "Les blocs de longueurs doivent etre de 3 elements");
080            }
081            
082            final long[] lCopieLongueurs = lLongeurs.clone();
083            Arrays.sort(lCopieLongueurs);
084            
085            final long lLongueur1 = lCopieLongueurs[0] * lCopieLongueurs[0];
086            final long lLongueur2 = lCopieLongueurs[1] * lCopieLongueurs[1];
087            final long lLongueur3 = lCopieLongueurs[2] * lCopieLongueurs[2];
088            
089            if(lLongueur1 + lLongueur2 == lLongueur3) {
090                lRetours[i] = true;
091            }
092            else {
093                lRetours[i] = false;
094            }
095        }
096
097        return lRetours;
098    }
